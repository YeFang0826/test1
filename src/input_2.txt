@ pre: 0<= l and u<|a| and sorted(a, l, u)
@ post: rv <=> (exist(int i), l<=i and i<=u and a[i]==e )
boolean BinarySearch(int[] a, int l, int u, int e){
     if(l>u){
        return false;
     }
     else{
        int m = (l+u)/2;
        if(a[m]==e){
         	return true;
        }
        elif(a[m]<e){
           return BinarySearch(a, m+1, u, e);
        }
        else {
           return BinarySearch(a, l, m-1, e);
        }
     }
}
